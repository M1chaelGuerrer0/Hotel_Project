77       /* 
78           Fires on the scene's startup, displaying the user's account information on the Account scene 
79       */
80       public void displayInfo() {
81           User guest;
82   
83           if (inputEmail.contains(empFlag) || inputEmail.contains(manFlag)) {
84               guest = HotelDataBase.getWorker(inputEmail);
85           } else { guest = HotelDataBase.getGuest(inputEmail); }
86   
87           assert guest != null;
88           String firstName = guest.getFirst_Name();
89           String lastName = guest.getLast_Name();
90           String email = guest.getEmail();
91           String password = guest.getPassword();
92           String phoneNum = guest.getPhone_Number();
93           String address1 = guest.getAddress1();
94           String address2 = guest.getAddress2();
95           String zip = guest.getZip_Code();
96           String state = guest.getState();
97           String country = guest.getCountry();
98           String city = guest.getCity();
99   
100          firstDisplay.setText(firstName);
101          lastDisplay.setText(lastName);
102          emailDisplay.setText(email);
103          passwordDisplay.setText(password);
104          phoneDisplay.setText(phoneNum);
105          address1Display.setText(address1);
106          address2Display.setText(address2);
107          zipDisplay.setText(zip);
108          stateDisplay.setText(state);
109          countryDisplay.setText(country);
110          cityDisplay.setText(city);
111      }

113      /* 
114          Switches from the current scene back to the Parent scene 
115   
116          @param event listens for when an event fires 
117      */
118      public void cancelButton(ActionEvent event) throws IOException {
119          root = FXMLLoader.load(getClass().getResource("parent.fxml"));
120          stage = (Stage)((Node)event.getSource()).getScene().getWindow();
121          scene = new Scene(root);
122          stage.setScene(scene);
123          stage.show();
124      }
