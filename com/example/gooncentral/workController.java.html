100      /* 
101          Refresh button refreshes the table for any changes made 
102          pressed when the manager wants to clear workers that have been fired/hired 
103      */
104      @FXML
105      public void Refresh() {
106          table.setItems(FXCollections.observableArrayList(getWorkers()));
107      }
108  
109      /* 
110          Delete button deletes selected worker 
111          pressed when the manager wants to fire a worker 
112      */
113  
114      @FXML
115      public void Delete() {
116          User selected = table.getSelectionModel().getSelectedItem();
117          if (selected != null) {
118              String mail = selected.getEmail();
119              HotelDataBase.deleteWorker(mail);
120              table.refresh();
121          }
122      }
123  
124      /* 
125          Switches from the current scene back to the Login scene 
126   
127          @param event listens for when an event fires 
128      */
129      public void cancelButton(ActionEvent event) throws IOException {
130          root = FXMLLoader.load(getClass().getResource("Parent.fxml"));
131          stage = (Stage)((Node)event.getSource()).getScene().getWindow();
132          scene = new Scene(root);
133          stage.setScene(scene);
134          stage.show();
135      }
