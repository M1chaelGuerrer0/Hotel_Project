118      public String loginButton(ActionEvent event) throws IOException {
119          inputEmail = userEmail.getText();                   // fetches input email in login text field for email
120          String inputPassword = userPassword.getText();      // fetches input password in login text field for password
121  
122          User guest;
123  
124          /* 
125              Verifies that the user's entered data matches with info found in the database, first verifying the 
126              email exists by retrieving it from the database as an object, and then comparing the password input 
127              to the one in the database. Catches runtime exceptions when the email could not be found. 
128          */
129          if (inputEmail.contains(empFlag) || inputEmail.contains(manFlag)) {
130  
131              empStatus = true;
132              System.out.println("Employee logged in: " + empStatus);
133  
134              manStatus = inputEmail.contains(manFlag);
135              System.out.println("Manager logged in: " + manStatus);
136  
137              try {
138                  guest = HotelDataBase.getWorker(inputEmail);
139                  password = guest.getPassword();
140  
141                  int match = inputPassword.compareTo(password);         // compares the two password input to confirm if they match
142                  System.out.println("Password match status: " + match); // will return 0, if a mismatch will return an int != 0
143  
144                  /* 
145                      Checks whether match was 0, and if so then the user may proceed to the Parent scene, and if not mismatch() 
146                      is called and the user is prompted to check their entered information 
147                  */
148                  if (match == 0) {
149                      logStatus = true;
150                      System.out.println("logStatus: " + logStatus);
151                      System.out.println("Your password is correct.");
152                      root = FXMLLoader.load(getClass().getResource("Parent.fxml"));
153                      stage = (Stage)((Node)event.getSource()).getScene().getWindow();
154                      scene = new Scene(root);
155                      stage.setScene(scene);
156                      stage.show();
157                  } else {
158                      System.out.println("Your password did not match.");
159                      mismatch();
160                  }
161              } catch (RuntimeException e) {
162                  mismatch();
163              }
164          } else {
165  
166              try {
167                  guest = HotelDataBase.getGuest(inputEmail);
168                  password = guest.getPassword();
169  
170                  int match = inputPassword.compareTo(password);         // compares the two password input to confirm if they match
171                  System.out.println("Password match status: " + match); // will return 0, if a mismatch will return an int != 0
172  
173              /* 
174                  Checks whether match was 0, and if so then the user may proceed to the Parent scene, and if not mismatch() 
175                  is called and the user is prompted to check their entered information 
176              */
177                  if (match == 0) {
178                      logStatus = true;
179                      System.out.println("logStatus: " + logStatus);

182                      stage = (Stage) ((Node) event.getSource()).getScene().getWindow();
183                      scene = new Scene(root);

191                  mismatch();
